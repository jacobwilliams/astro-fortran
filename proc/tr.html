    subroutine TR ( r, rt )

    implicit none

    real(wp),dimension(3,3),intent(in) :: r !! r-matrix
    real(wp),dimension(3,3),intent(out) :: rt !! transpose

    real(wp) :: wm(3,3)
    integer :: i, j

    do i=1,3
       do j=1,3
          wm(i,j) = r(j,i)
       end do
    end do
    call CR ( wm, rt )

    end subroutine TR
