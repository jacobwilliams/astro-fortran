    subroutine S2C ( theta, phi, c )

    implicit none

    real(wp),intent(in) :: theta !!  longitude angle (radians)
    real(wp),intent(in) :: phi !!  latitude angle (radians)
    real(wp),dimension(3),intent(out) :: c !! direction cosines

    real(wp) :: cp

    cp = cos(phi)
    c(1) = cos(theta) * cp
    c(2) = sin(theta) * cp
    c(3) = sin(phi)

    end subroutine S2C
