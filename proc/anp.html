    function ANP ( a ) result(w)

    implicit none

    real(wp),intent(in) :: a !! angle (radians)
    real(wp) :: w !! angle in range 0-2pi

    w = mod(a,d2pi)
    if ( w < 0.0_wp ) w = w + d2pi

    end function ANP
