    subroutine RXR ( a, b, atb )

    implicit none

    real(wp),dimension(3,3),intent(in) :: a !! first r-matrix
    real(wp),dimension(3,3),intent(in) :: b !! second r-matrix
    real(wp),dimension(3,3),intent(out) :: atb !! A * B

    integer :: i, j, k
    real(wp) :: w, wm(3,3)

    do i=1,3
       do j=1,3
          w = 0.0_wp
          do k=1,3
             w = w + a(i,k)*b(k,j)
          end do
          wm(i,j) = w
       end do
    end do
    call CR ( wm, atb )

    end subroutine RXR
