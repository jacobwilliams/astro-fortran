    subroutine IR ( r )

    implicit none

    real(wp),dimension(3,3),intent(out) :: r !! r-matrix

    r(1,1) = 1.0_wp
    r(1,2) = 0.0_wp
    r(1,3) = 0.0_wp
    r(2,1) = 0.0_wp
    r(2,2) = 1.0_wp
    r(2,3) = 0.0_wp
    r(3,1) = 0.0_wp
    r(3,2) = 0.0_wp
    r(3,3) = 1.0_wp

    end subroutine IR
