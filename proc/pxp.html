    subroutine PXP ( a, b, axb )

    implicit none

    real(wp),dimension(3),intent(in) :: a !! first p-vector
    real(wp),dimension(3),intent(in) :: b !! second p-vector
    real(wp),dimension(3),intent(out) :: axb !! A x B

    real(wp) :: xa, ya, za, xb, yb, zb

    xa = a(1)
    ya = a(2)
    za = a(3)
    xb = b(1)
    yb = b(2)
    zb = b(3)
    axb(1) = ya*zb - za*yb
    axb(2) = za*xb - xa*zb
    axb(3) = xa*yb - ya*xb

    end subroutine PXP
