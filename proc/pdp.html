    subroutine PDP ( a, b, adb )

    implicit none

    real(wp),dimension(3),intent(in) :: a !! first p-vector
    real(wp),dimension(3),intent(in) :: b !! second p-vector
    real(wp),intent(out) :: adb !! A . B

    real(wp) :: w
    integer :: i

    w = 0.0_wp
    do i=1,3
       w = w + a(i)*b(i)
    end do
    adb = w

    end subroutine PDP
