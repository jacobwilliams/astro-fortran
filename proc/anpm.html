    function ANPM ( a ) result(w)

    implicit none

    real(wp),intent(in) :: a !! angle (radians)
    real(wp) :: w !! angle in range +/-pi

    w = mod(a,d2pi)
    if ( abs(w) >= dpi ) w = w - sign(d2pi,a)

    end function ANPM
