    subroutine PMP ( a, b, amb )

    implicit none

    real(wp),dimension(3),intent(in) :: a !! first p-vector
    real(wp),dimension(3),intent(in) :: b !! second p-vector
    real(wp),dimension(3),intent(out) :: amb !! A - B

    integer :: i

    do i=1,3
       amb(i) = a(i) - b(i)
    end do

    end subroutine PMP
