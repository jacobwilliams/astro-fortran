    subroutine TRXP ( r, p, trp )

    implicit none

    real(wp),dimension(3,3),intent(in) :: r !! r-matrix
    real(wp),dimension(3),intent(in) :: p !! p-vector
    real(wp),dimension(3),intent(out) :: trp !! R * P

    real(wp) :: ri(3,3)

    !  Transpose of matrix R.
    call TR ( r, ri )

    !  Matrix RI * vector P -> vector TRP.
    call RXP ( ri, p, trp )

    end subroutine TRXP
